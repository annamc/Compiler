. . . a simple parser for a simple grammar

Program       ::== Statement $
Statement     ::== P ( Expr )
	      ::== Id = Expr
	      ::== VarDecl
	      ::== { StatementList }
StatementList ::== Statement StatementList
	      ::== epsilon
Expr          ::== IntExpr
	      ::== CharExpr
	      ::== Id
IntExpr       ::== digit op Expr
	      ::== digit
CharExpr      ::== " CharList "
CharList      ::== Char CharList
	      ::== epsilon
VarDecl       ::== Type Id
Type          ::== int | char
Id            ::== Char
Char          ::== a | b | c ... z
digit         ::== 1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 | 9 | 0
op            ::== + | -
