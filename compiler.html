. . . a simple parser for a simple grammar

Program       ::== Statement $
Statement     ::== print ( Expr )
              ::== Id = Expr
              ::== VarDecl
              ::== { StatementList }
StatementList ::== Statement StatementList
              ::== epsilon
Expr          ::== IntExpr
              ::== StringExpr
              ::== Id
IntExpr       ::== digit op Expr
              ::== digit
StringExpr    ::== " CharList "
CharList      ::== Char CharList
              ::== Space CharList
              ::== epsilon
VarDecl       ::== Type Id
Type          ::== int | string
Id            ::== Char
Char          ::== a | b | c ... z
Space         ::== the space character
digit         ::== 1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 | 9 | 0
op            ::== + | -
